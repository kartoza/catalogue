 77 /** A helper function for jquery sortable to prevent rows collapsing
 78  ** the cell width when dragging them.
 79  ** @see http://www.foliotek.com/devblog/make-table-rows-sortable-using-jquery-ui-sortable/
 80  ** @note: to use do
 81  **   $("#sort tbody").sortable({ 
 82  **     helper: sortableHelperFix
 83  **   }).disableSelection();
 84  **/
 85 var sortableHelperFix = function(e, ui) {
 86   ui.children().each(function() {
 87     $(this).width($(this).width());
 88   });
 89   return ui;
 90 };

106 /** Helper function to remove an element from an array
107  * Array Remove - By John Resig (MIT Licensed)
108  * @see http://ejohn.org/blog/javascript-array-remove/
109  */
110 function removeElements(theArray, from, to) {
111  var rest = theArray.slice((to || from) + 1 || theArray.length);
112  theArray.length = from < 0 ? theArray.length + from : from;
113  return theArray.push.apply(theArray, rest);
114 };

117  * see http://code.google.com/p/humanmsg/

120 /*
121   HUMANIZED MESSAGES 1.0
122   idea - http://www.humanized.com/weblog/2006/09/11/monolog_boxes_and_transparent_messages
123   home - http://humanmsg.googlecode.com
124   MODIFIED VERSION by Linfiniti CC, November 2011

127 var humanMsg = {
128   setup: function(appendTo, logName, msgOpacity) {
129     humanMsg.msgID = 'humanMsg';
130     humanMsg.logID = 'humanMsgLog';
131 
132     // appendTo is the element the msg is appended to
133     if (appendTo == undefined)
134       appendTo = 'body';
135 
136     // The text on the Log tab
137     if (logName == undefined)
138       logName = 'Message Log';
139 
140     // Opacity of the message
141     humanMsg.msgOpacity = .8;
142 
143     if (msgOpacity != undefined) 
144       humanMsg.msgOpacity = parseFloat(msgOpacity);
145 
146     // Inject the message structure
147     $(appendTo).append('<div id="'+humanMsg.msgID+'" class="humanMsg"><div class="round"></div><p></p><div class="round"></div></div> <div id="'+humanMsg.logID+'"><p id="humanMsgLogTitle"><b>'+logName+'</b></p><ul></ul></div>');
148 
149     $('#'+humanMsg.logID+' p').click(
150       function() { $(this).siblings('ul').slideToggle() }
151     )
152   },
153 
154   displayMsg: function(msg, msgType) {
155     if (msg == '')
156       return;
157 
158     clearTimeout(humanMsg.t2);
159 
160     // Inject message
161     $('#'+humanMsg.msgID+' p').html(msg);
162 
163     // Set the appropriate class: error "e", or normal "n".
164     if (msgType == "n")
165     {
166       $("#"+humanMsg.msgID).removeClass("humanMsgError");
167       $("#"+humanMsg.msgID).addClass("humanMsgNormal");
168     }
169     else
170     {
171       $("#"+humanMsg.msgID).removeClass("humanMsgNormal");
172       $("#"+humanMsg.msgID).addClass("humanMsgError");
173     }
174 
175     // Show message
176     $('#'+humanMsg.msgID+'').show().animate({ opacity: humanMsg.msgOpacity}, 200, function() {
177       $('#'+humanMsg.logID)
178         .show().children('ul').prepend('<li>'+msg+'</li>')  // Prepend message to log
179         .children('li:first').slideDown(200)                // Slide it down
180 
181       if ( $('#'+humanMsg.logID+' ul').css('display') == 'none') {
182         $('#'+humanMsg.logID+' p').addClass("ui-state-active");
183         $('#'+humanMsg.logID+' p').animate({ bottom: 40 }, 200, 'linear', function() {
184           $(this).animate({ bottom: 0 }, 300, 'easeOutBounce', function() { $(this).css({ bottom: 0 }) })
185         });
186       }
187     })
188 
189     // Watch for mouse & keyboard in .5s
190     humanMsg.t1 = setTimeout("humanMsg.bindEvents()", 700)
191     // Remove message after 5s
192     // humanMsg.t2 = setTimeout("humanMsg.removeMsg()", 5000)
193   },
194 
195   bindEvents: function() {
196   // Remove message if mouse is moved or key is pressed
197     $(window)
198       .mousemove(humanMsg.removeMsg)
199       .click(humanMsg.removeMsg)
200       .keypress(humanMsg.removeMsg)
201   },
202 
203   removeMsg: function() {
204     // Unbind mouse & keyboard
205     $(window)
206       .unbind('mousemove', humanMsg.removeMsg)
207       .unbind('click', humanMsg.removeMsg)
208       .unbind('keypress', humanMsg.removeMsg)
209 
210     // fade out the message
211     $('#'+humanMsg.msgID).delay(1000).animate({ opacity: 0 }, 500, function() { $(this).hide() })
212   }
213 };

249 /** This function is the first thing called when the page loads.
250  *  @param theExtent must be passed as an OpenLayers Bounds object in Geographic projection.
251  *  @param theOverlayMapUrl must be a WMS URL.
252  *  @param theOverlayLayer the name of the layer in the WMS GetCapabilities document.
253  */
254 function initialisePrintMap( theOverlayMapUrl, theOverlayLayer, theExtent )
255 {
256   window.status="working";
257   mGeoProjection = new OpenLayers.Projection("EPSG:4326");
258 
259   mMercatorProjection = new OpenLayers.Projection("EPSG:900913");
260 
261   var myExtent = theExtent.transform(
262                mGeoProjection, mMercatorProjection
263              );
264   var myCenterPoint = new OpenLayers.LonLat(25,-24).transform(
265                mGeoProjection, mMercatorProjection
266              );
267   var myOptions = {
268                projection: mMercatorProjection,
269                displayProjection: mGeoProjection,
270                units: "m",
271                maxResolution: 156543.0339,
272                maxExtent: myExtent,
273                controls: [
274                ]
275              };
276   var myMap = new OpenLayers.Map('map', myOptions);
277 
278   var myGoogleHybridLayer = new OpenLayers.Layer.Google(
279                "Google Hybrid",
280                { type: google.maps.MapTypeId.HYBRID, numZoomLevels: 20}
281              );
282   var myOverlay = new OpenLayers.Layer.WMS( "Overlay",
283       theOverlayMapUrl,
284       {
285         layers: theOverlayLayer,
286         format: 'image/gif',
287         //format: 'image/png; mode=24bit',
288         version: '1.3'
289       },
290       {
291         singleTile: true,
292         isBaseLayer: false,
293         visibility: true
294       });
295 
296   myMap.addLayers([myGoogleHybridLayer, myOverlay]);
297   myMap.zoomToExtent(myExtent);
298   window.status="ready";
299 }
